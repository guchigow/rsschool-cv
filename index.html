 function validParentheses(parens) {
   let open, close;
   close = open = 0;
   for(let i = 0; i < parens.length; i++) {
     if(open < close) return false;
     if(parens[i] === "(") open++;
     if(parens[i] === ")") close++;
   }
   return open === close;
 }
